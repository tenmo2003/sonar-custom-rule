This rule reminds team members to use BillPugh implementation when using singleton.

Noncompliant Code Example

    
public final class SingletonExample {

    private static SingletonExample INSTANCE;

    private SingletonExample() {
    }

    public static SingletonExample getInstance() {
        if(INSTANCE == null) {
            INSTANCE = new SingletonExample();
        }

        return INSTANCE;
    }

    // getters and setters
}

Compliant Solution

    
public class SingletonExample {
    private SingletonExample() {

    }

    private static class SingletonHelper {
        private static final SingletonExample BILL_PUGH_SINGLETON_INSTANCE = new SingletonExample();
    }

    public static SingletonExample getInstance() {
        return SingletonHelper.BILL_PUGH_SINGLETON_INSTANCE;
    }
}
